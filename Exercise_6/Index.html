Write a function that performs the bubble algorithm.

Input [3, 6, 12, 5, 100, 1]

Output [1, 3, 5, 6, 12, 100]


const numeros = [3, 6, 12, 5, 100, 1];

const length = numeros.length;

for (let i = 0; i < length; i++) {

for (let j = 0; j < (length - i - 1); j++) {

if (numeros[j] > numeros[j + 1]) {

const temp = numeros[j];

numeros[j] = numeros[j + 1];

numeros[j + 1] = temp;

}

}

}
console.log("Usando la funcion burbuja el arreglo queda:");

console.log(numeros);
